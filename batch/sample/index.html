class BatchSample:
    def test_all(self):
        app_all = ["mayapy", "rez", "mobupy", "motionbuilder", "maya"] #, "3dsmax", "3dsmaxpy"] # ["motionbuilder"] #, "mobupy", "motionbuilder", "maya", "mayapy", "rez"] #, "3dsmax", "3dsmaxpy"
        for app in app_all:
            versions = ["2020"] #["2018", "2019", "2020", "2022", "2023"]
            for version in versions:
                return_codes = self.execute_one(app, version, False)
                if return_codes:
                    # self.assertEqual(return_codes, [0, 0, 0])
                    if return_codes == [0, 0, 0]:
                        print("OK")
                    else:
                        print("NG: {}".format(return_codes))
                else:
                    print("job failed")

def execute_one(self, app, version, file_mode=False):
    # app: maya, mayapy, rez, motionbuilder, mobupy

    command_data = {"version": version}

    task_set = [{"step": "pre", "tasks": [{"module": "tasks.win.open_file"}]},
                {"step": "main", "tasks": [{"module": "tasks.win.export_file"}]}]

    data_set = {"pre": {"open_path": "somewhere"}, 
            "main": [{"name": "nameA"}, {"name": "namessB"}]}

    command_data["sys_path"] = "sys path for ..."
    command_data["module_directory_path"] = module_path
    command_data["job_directory"] = "temp directory for job"
    if file_mode:
        command_data["task_set_path"] = "task yml file"
        command_data["data_path"] = "data json file"
    else:
        command_data["task_set"] = task_set
        command_data["data_set"] = data_set

    command_data["close_app"] = True
    if app == "rez":
        command_data["launcher"] = "rez"
        command_data["rez_package"] = "projectA"
        command_data["rez_version"] = "1.0.0"
        command_data["app_package"] = "mayapy"
        app = "mayapy"

    command, job_directory = run_process(app, **command_data)
    result_path = "{}/results.json".format(job_directory)

    print("result_path: {}".format(result_path))
    if command:
        try:
            info, data = pz_config.read(result_path)
            print("data:")
            pprint.pprint(data)
        except:
            print("read failed")
            import traceback
            traceback.print_exc()
        return [l["return_code"] for l in data]

    return False

if __name__ == "__main__":
    x = BatchSample()

    x.execute_one("mayapy", "2020")
